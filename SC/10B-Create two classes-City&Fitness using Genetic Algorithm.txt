import math
import random

class City:
  def __init__(self, x=None, y=None):
    self.x=x if x is not None else random.randint(0,100)
    self.y=y if y is not None else random.randint(0,100)

  def distance(self, city):
   return math.sqrt((self.x-city.x)**2 + (self.y-city.y)**2)

  def __repr__(self):
    return f"({self.x},{self.y})"

class Fitness:
  def __init__(self, route):
    self.route=route
    self.distance_value=0
    self.fitness_value=0.0

  def route_distance(self):
    if self.distance_value==0:
      path_distance=0
      for i in range(len(self.route)):
        start=self.route[i]
        end=self.route[(i+1)%len(self.route)]
        path_distance+=start.distance(end)
      self.distance_value=path_distance
    return self.distance_value

  def route_fitness(self):
    if self.fitness_value==0:
       self.fitness_value=1 / float(self.route_distance())
    return self.fitness_value

cities = [City(), City(), City(), City(), City()]

print("Cities:", cities)
fit = Fitness(cities)
print("Distance:", fit.route_distance())
print("Fitness:", fit.route_fitness())
