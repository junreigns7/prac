# Take input values for x (input), w (weight), and b (bias)
x=float(input("Enter value of x."))
w=float(input("Enter value of weight w. "))
b=float(input("Enter value of bias b."))
# Calculate the net input (weighted sum + bias)
net=int(w*x+b)
# Apply a piecewise linear activation function
if (net<0):
  out=0
elif((net>=0)&(net<=1)):
  out=net
else:
  out=1
# Print results
  print("net=",net)
  print("output=",out)
