import math
n=int(input("Enter number of elements:"))
print("enter the inputs:")
inputs=[]
for i in range(n):
  ele=float(input())
  inputs.append(ele)
print("Inputs:",inputs)
print("enter the weights:")
weights=[]
for i in range(n):
  w=float(input())
  weights.append(w)
print("Weights:",weights)
Yin=[]
for i in range(n):
  Yin.append(inputs[i]*weights[i])
net_input=sum(Yin)
print("\nNet input (Yin)=",round(net_input, 3))
binary_sigmoid=1/(1+math.exp(-net_input))
print("Binary Sigmoid Output=",round(binary_sigmoid, 3))
bipolar_sigmoid=(1-math.exp(-net_input)) / (1+math.exp(-net_input))
print("Bipolar Sigmoid Output=", round(bipolar_sigmoid, 3))
